STEP 1: Download ZIP from GitHub
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Go to: https://github.com/redinc23/letters-app
2. Click green "Code" button
3. Click "Download ZIP"
4. Save the file (usually downloads as letters-app-main.zip)

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 2: Create New Repl
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. Go to: https://replit.com
2. Click "Create Repl"
3. Choose "Node.js" template
4. Name: letters-app
5. Click "Create"

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 3: Upload ZIP File
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. In Replit file explorer, click three dots (⋯)
2. Click "Upload file"
3. Select the downloaded ZIP file
4. Wait for upload to complete

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 4: Extract ZIP
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

In Replit shell, run:

unzip letters-app-main.zip -d .
mv letters-app-main/* .
mv letters-app-main/.* . 2>/dev/null || true
rmdir letters-app-main
rm letters-app-main.zip

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

STEP 5: Run Setup
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

bash DO_EVERYTHING.sh

Then add the 3 secrets it shows you. Done!
